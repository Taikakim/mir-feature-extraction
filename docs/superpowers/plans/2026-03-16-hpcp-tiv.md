> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a parallel HPCP/TIV/tonic harmonic feature pipeline alongside the existing CQT chroma, using Essentia's peak-based pitch extraction and TIV projection for more accurate conditioning signals.

**Architecture:** New self-contained module `src/harmonic/hpcp_tiv.py` computes HPCP via Essentia's SpectralPeaks pipeline, derives TIV via weighted DFT, and estimates tonic via Essentia's Key algorithm — all from a bass+other stem mix. The feature extractor gets a new `_extract_hpcp_tiv()` method alongside the existing chroma step, which is also fixed to use bass+other instead of other-only.

**Tech Stack:** Essentia (SpectralPeaks, HPCP, Key), NumPy (DFT/normalization), librosa (not used in new module — Essentia only), soundfile (stem loading), pytest (unit tests).

```python
"""
HPCP + TIV + Tonic Harmonic Feature Extraction

Parallel to CQT chroma (chroma_0..11). Uses Essentia's HPCP (peak-picking +
harmonic summation) instead of librosa's direct bin folding — resistant to
overtone contamination and separation artifacts.

TIV (Tonal Interval Vector): DFT of HPCP with perceptual weights. Averaging
the TIV over 10s yields a meaningful measure of harmonic tension/diatonicity
rather than a smeared tonal-mass blur.

Stem selection: bass + other (no vocals — pitch instability smears bins;
no drums — transient broadband noise). Falls back to full mix with a warning.

Dependencies: essentia, numpy, soundfile
"""

import logging
import sys
from pathlib import Path
from typing import Dict, Optional, Tuple

import numpy as np
import soundfile as sf

sys.path.insert(0, str(Path(__file__).parent.parent))

from core.file_utils import get_stem_files, read_audio
from core.common import clamp_feature_value
from core.json_handler import safe_update, get_info_path

logger = logging.getLogger(__name__)

PITCH_CLASSES = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B']

# Perceptual weights from Harte et al. 2006 — calibrated to human consonance ratings
_TIV_WEIGHTS = np.array([3.0, 8.0, 11.5, 15.0, 14.5, 7.5])


def mix_bass_other_stems(
    folder_path: Path,
) -> Tuple[Optional[np.ndarray], int, bool]:
    """
    Mix bass + other stems (excluding drums and vocals) for harmonic analysis.

    Falls back to full mix with a warning if neither stem is available.

    Returns:
        (audio_mono, sample_rate, used_stems)
        used_stems is False when falling back to full mix.
    """
    stems = get_stem_files(folder_path, include_full_mix=True)
    target = [s for s in ['bass', 'other'] if s in stems]

    if len(target) >= 1:
        logger.info(f"Using harmonic stems: {', '.join(target)} (excluding drums and vocals)")
        mixed: Optional[np.ndarray] = None
        sr: int = 0
        for name in target:
            audio, stem_sr = sf.read(str(stems[name]), dtype='float32')
            if audio.ndim > 1:
                audio = audio.mean(axis=1)
            if mixed is None:
                mixed, sr = audio, stem_sr
            else:
                max_len = max(len(mixed), len(audio))
                mixed = np.pad(mixed, (0, max_len - len(mixed)))
                audio = np.pad(audio, (0, max_len - len(audio)))
                mixed = mixed + audio
        max_val = float(np.abs(mixed).max())
        if max_val > 0:
            mixed = mixed / max_val * 0.95
        return mixed, sr, True

    if 'full_mix' in stems:
        logger.warning(
            "HPCP: bass and other stems not available — falling back to full mix. "
            "Drum transients will degrade pitch extraction quality."
        )
        audio, sr = read_audio(str(stems['full_mix']), dtype='float32')
        if audio.ndim > 1:
            audio = audio.mean(axis=1)
        return audio, sr, False

    return None, 0, False


def compute_hpcp_frames(
    audio: np.ndarray,
    sr: int,
    frame_size: int = 4096,
    hop_size: int = 2048,
) -> np.ndarray:
    """
    Compute per-frame HPCP using Essentia's peak-picking pipeline.

    HPCP uses spectral peak detection + harmonic summation to suppress
    overtone contamination that corrupts direct bin-folding methods (CQT).

    Returns:
        np.ndarray shape (n_frames, 12). May be empty if audio is very short.
    """
    import essentia.standard as es

    windowing = es.Windowing(type='hann', size=frame_size)
    spectrum_alg = es.Spectrum(size=frame_size)
    spectral_peaks = es.SpectralPeaks(
        maxPeaks=100,
        magnitudeThreshold=0.00001,
        sampleRate=sr,
        orderBy='magnitude',
    )
    hpcp_alg = es.HPCP(
        size=12,
        harmonics=8,
        minFrequency=40.0,
        maxFrequency=5000.0,
        weightType='cosine',
        nonLinear=True,
        normalized='unitMax',
        referenceFrequency=440.0,
        sampleRate=sr,
    )

    frames = []
    for frame in es.FrameGenerator(audio, frameSize=frame_size, hopSize=hop_size):
        windowed = windowing(frame)
        spec = spectrum_alg(windowed)
        freqs, mags = spectral_peaks(spec)
        hpcp_frame = hpcp_alg(freqs, mags)
        frames.append(hpcp_frame)

    if not frames:
        return np.zeros((0, 12), dtype=np.float32)
    return np.array(frames, dtype=np.float32)


def compute_tiv(hpcp_avg: np.ndarray) -> np.ndarray:
    """
    Compute Tonal Interval Vector from an averaged HPCP vector.

    Projects the 12-bin pitch-class distribution into a perceptually motivated
    space where each dimension encodes a specific intervallic property
    (chromaticity, minor thirds, major thirds, dissonance, diatonicity,
    whole-toneness). Averaging TIV over time yields semantic harmonic properties
    rather than a smeared tonal mass.

    Args:
        hpcp_avg: 12-element averaged HPCP vector (any normalization)

    Returns:
        12-element L2-normalized TIV: [Re(k1), Im(k1), ..., Re(k6), Im(k6)]
        Returns all-zeros if input is silent.
    """
    total = float(hpcp_avg.sum())
    if total == 0.0:
        return np.zeros(12, dtype=np.float64)

    # L1-normalize (remove DC / total energy)
    c = hpcp_avg / total

    # DFT of the 12-bin pitch class profile
    dft = np.fft.fft(c)

    # Apply perceptual weights to k=1..6, flatten to 12 real values
    tiv_flat = np.zeros(12, dtype=np.float64)
    for k in range(1, 7):
        weighted = _TIV_WEIGHTS[k - 1] * dft[k]
        tiv_flat[2 * (k - 1)]     = weighted.real
        tiv_flat[2 * (k - 1) + 1] = weighted.imag

    # L2-normalize to unit hypersphere
    norm = float(np.linalg.norm(tiv_flat))
    if norm == 0.0:
        return np.zeros(12, dtype=np.float64)
    return tiv_flat / norm
```

def analyze_hpcp_tiv(
    audio_path,
    use_stems: bool = True,
    audio: Optional[np.ndarray] = None,
    sr: Optional[int] = None,
) -> Dict:
    """
    Analyze HPCP, TIV, and tonic features for an audio file.

    When audio is provided, stem loading is skipped entirely (use_stems ignored).
    Pass use_stems=False as a defensive convention when providing audio.

    Returns dict with:
        hpcp_0..11  — L∞-normalized pitch class weights [0,1]
        tiv_0..11   — L2-normalized Tonal Interval Vector [-1,1]
        tonic       — estimated root pitch class (0=C..11=B)
        tonic_strength — key detection confidence [0,1]
        tonic_scale — "major", "minor", or "unknown"
    """
    import essentia.standard as es

    audio_path = Path(audio_path)

    if audio is not None:
        # Pre-loaded audio provided — skip all disk/stem logic
        logger.info(f"Analyzing HPCP+TIV: {audio_path.name} (pre-loaded)")
        used_stems = False
    else:
        if not audio_path.exists():
            raise FileNotFoundError(f"Audio file not found: {audio_path}")
        logger.info(f"Analyzing HPCP+TIV: {audio_path.name}")

        used_stems = False
        if use_stems and audio_path.parent.is_dir():
            audio, sr, used_stems = mix_bass_other_stems(audio_path.parent)

        if audio is None:
            audio, sr = read_audio(audio_path, dtype='float32')
            if audio.ndim > 1:
                audio = audio.mean(axis=1)

    if audio.ndim > 1:
        audio = audio.mean(axis=1)

    source_label = "bass+other stems" if used_stems else "full mix / pre-loaded"
    logger.info(f"  Analysis source: {source_label}")

    # --- Compute HPCP frames ---
    hpcp_frames = compute_hpcp_frames(audio, sr)

    # Silent / empty guard
    if hpcp_frames.shape[0] == 0 or hpcp_frames.sum() == 0:
        logger.warning(f"  Silent or empty audio — returning zero HPCP/TIV for {audio_path.name}")
        result = {}
        for i in range(12):
            result[f'hpcp_{i}'] = 0.0
            result[f'tiv_{i}']  = 0.0
        result['tonic']          = 0.0
        result['tonic_strength'] = 0.0
        result['tonic_scale']    = 'unknown'
        return result

    # Average HPCP frames → 12-dim vector
    hpcp_avg = hpcp_frames.mean(axis=0).astype(np.float64)

    # Final L∞ normalize on averaged vector
    hpcp_max = float(hpcp_avg.max())
    if hpcp_max > 0:
        hpcp_avg = hpcp_avg / hpcp_max

    # Compute TIV
    tiv = compute_tiv(hpcp_avg)

    # --- Tonic estimation via Essentia Key ---
    tonic_int = 0
    tonic_strength = 0.0
    tonic_scale = 'unknown'
    try:
        try:
            key_alg = es.Key(profileType='edmm')
        except Exception:
            logger.warning("Essentia Key: 'edmm' profile unavailable, falling back to 'edma'")
            key_alg = es.Key(profileType='edma')

        # Key returns (key, scale, strength, firstToSecondRelativeStrength)
        # 4th value is a ratio that can exceed 1.0 — intentionally discarded
        key_str, scale_str, strength, _ = key_alg(hpcp_avg.astype(np.float32))
        if key_str in PITCH_CLASSES:
            tonic_int = PITCH_CLASSES.index(key_str)
        else:
            logger.warning(f"  Key algorithm returned unknown pitch class: '{key_str}', defaulting to 0")
        tonic_strength = float(np.clip(strength, 0.0, 1.0))
        tonic_scale = scale_str
    except Exception as e:
        logger.warning(f"  Tonic estimation failed: {e}")

    # --- Assemble result ---
    result = {}
    for i in range(12):
        result[f'hpcp_{i}'] = clamp_feature_value(f'hpcp_{i}', float(hpcp_avg[i]))
        result[f'tiv_{i}']  = clamp_feature_value(f'tiv_{i}',  float(tiv[i]))
    result['tonic']          = clamp_feature_value('tonic', float(tonic_int))
    result['tonic_strength'] = clamp_feature_value('tonic_strength', tonic_strength)
    result['tonic_scale']    = tonic_scale  # string, not clamped

    # Log HPCP distribution
    logger.info("HPCP pitch class distribution (L∞ normalized):")
    for i in range(12):
        weight = result[f'hpcp_{i}']
        bar = '█' * int(weight * 40)
        logger.info(f"  {PITCH_CLASSES[i]:>2}: {weight:.3f} {bar}")
    logger.info(f"  Tonic: {PITCH_CLASSES[tonic_int]} {tonic_scale} "
                f"(strength={tonic_strength:.3f})")

    return result


def batch_analyze_hpcp_tiv(
    root_directory,
    overwrite: bool = False,
) -> dict:
    """
    Batch analyze HPCP+TIV+tonic for all organized folders.

    Args:
        root_directory: Root directory to search
        overwrite: Whether to overwrite existing data

    Returns:
        Stats dict with total/success/skipped/failed/errors keys.
    """
    from core.file_utils import find_organized_folders
    from core.json_handler import should_process

    root_directory = Path(root_directory)
    logger.info(f"Starting batch HPCP+TIV analysis: {root_directory}")

    folders = find_organized_folders(root_directory)
    HPCP_KEYS = [f'hpcp_{i}' for i in range(12)]

    stats = {'total': len(folders), 'success': 0, 'skipped': 0,
             'failed': 0, 'errors': []}
    logger.info(f"Found {stats['total']} organized folders")

    for idx, folder in enumerate(folders, 1):
        logger.info(f"Processing {idx}/{stats['total']}: {folder.name}")
        stems = get_stem_files(folder, include_full_mix=True)
        if 'full_mix' not in stems:
            logger.warning(f"No full_mix found in {folder.name}")
            stats['failed'] += 1
            continue
        info_path = get_info_path(stems['full_mix'])
        if not should_process(info_path, HPCP_KEYS, overwrite):
            logger.info("  HPCP data already exists. Use --overwrite to regenerate.")
            stats['skipped'] += 1
            continue
        try:
            result = analyze_hpcp_tiv(stems['full_mix'], use_stems=True)
            safe_update(info_path, result)
            stats['success'] += 1
        except Exception as e:
            stats['failed'] += 1
            stats['errors'].append(f"{folder.name}: {e}")
            logger.error(f"  Failed: {e}")

    logger.info("=" * 60)
    logger.info(f"Batch HPCP+TIV: {stats['success']} ok, "
                f"{stats['skipped']} skipped, {stats['failed']} failed")
    logger.info("=" * 60)
    return stats
